import time
from collections import defaultdict

fail_count = defaultdict(int)       # 실패 횟수 추가
blocked_ips = set()                 # 차단 된 IP 저장할 공간

last_position = 0

def save_block(ip):
    with open("blocklist.txt", "a") as f:
        f.write(ip + "\n")

while True:
    with open("log.txt", "r") as f:
        f.seek(last_position)  # 이전 위치부터 읽기

        for line in f:
            parts = line.strip().split()

            # 🔥 안전장치 (필수)
            if len(parts) < 4:
                print("❌ 문제 줄 발견:", repr(line))
                continue

            status = parts[2]
            ip = parts[3]

            if status == "LOGIN_FAIL":
                fail_count[ip] += 1
                count = fail_count[ip]

                if count == 3:
                 print(f"⚠️ 주의: {ip} (3번)")
                elif count == 5:
                    print(f"🚨 위험: {ip} (5번)")
                elif count == 10:
                    print(f"🔥 매우 위험: {ip} (10번)")

                 # 🔥 차단 로직 추가
                    if ip not in blocked_ips:
                        print(f"⛔ 차단됨: {ip}")
                        save_block(ip)
                        blocked_ips.add(ip)    

        last_position = f.tell()  # 현재 위치 저장

    time.sleep(3)


# 결과 출력
for ip, count in fail_count.items():
    if count >= 3:
        print(f"⚠️ 의심 IP: {ip} (실패 {count}번)")
